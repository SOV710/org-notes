fish (Friendly Interactive SHell) 的设计哲学是开箱即用、语法清晰、避免历史包袱。我们直接从语言本身切入。

* Grammar

** Variable

#+begin_src fish
# 赋值用 set, 没有 $ 在左边
set name "world"
set count 42

# 读取用 $
echo $name
echo "hello $name"

# 列表 (fish 原生支持列表, 这是核心特性)
set fruits apple banana cherry
echo $fruits[1]        # apple (下标从 1 开始)
echo $fruits[2..]      # banana cherry (切片)
echo (count $fruits)   # 3
#+end_src

作用域修饰符:

#+begin_src fish
set -l local_var "只在当前作用域"   # local
set -g global_var "全局"            # global
set -x EXPORTED "导出为环境变量"    # export (等价于 export)
set -e var_to_delete                # 删除变量
#+end_src

注: 这是一个很大的坑点, 也是一个很大的难点, 后面我们会再讲到这个配置

** String

#+begin_src fish
# 双引号: 插值
set who "fish"
echo "hello $who"          # hello fish

# 单引号: 字面量, 无插值
echo 'hello $who'          # hello $who

# 字符串操作 (内建 string 命令, 非常强大)
string length "hello"                    # 5
string upper "hello"                     # HELLO
string replace "foo" "bar" "foo baz"     # bar baz
string match -r '\d+' "abc123"           # 123
string split "," "a,b,c"                 # a b c (列表)
string join "-" a b c                    # a-b-c
#+end_src

** Command Replacement

#+begin_src
# 用 () 而不是 $() 或反引号
set files (ls)
set today (date +%Y-%m-%d)
echo "today is $today"

# 直接嵌入
echo "lines: "(wc -l < /etc/hosts)
#+end_src

** Condition

#+begin_src fish
# if/else if/else/end (注意 end 结尾, 无大括号)
if test $count -gt 10
    echo "big"
else if test $count -gt 5
    echo "medium"
else
    echo "small"
end

# 字符串比较
if test "$name" = "fish"
    echo "it's fish"
end

# 文件测试
if test -f /etc/hosts
    echo "exists"
end
#+end_src

=[= 是 =test= 的 alias, 但 fish 社区更推荐直接使用 =test=

#+begin_src
# 命令成功/失败作为条件 (最惯用法)
if command -q git
    echo "git is installed"
end

if grep -q "pattern" file.txt
    echo "found"
end
#+end_src

** Loop

# for 循环 (自然遍历列表)
for fruit in $fruits
    echo "fruit: $fruit"

# 遍历数字范围 (seq)
for i in (seq 1 10)
    echo $i
end

    echo $count
    set count (math $count - 1)

** Functions

#+begin_src fish
# 用 function 定义, end 结尾
function greet
    echo "hello, $argv[1]"
end

greet world    # hello, world

# $argv 是参数列表
function sum
    set total 0
    for n in $argv
        set total (math $total + $n)
    end
    echo $total
end

sum 1 2 3 4    # 10
#+end_src

具名参数 (惯用法) :

#+begin_src
function connect
    set -l host $argv[1]
    set -l port $argv[2]
    echo "connecting to $host:$port"
end
#+end_src

=--argument-names= 语法糖 (fish 3.x) :

#+begin_src
function connect --argument-names host port
    echo "connecting to $host:$port"
end
#+end_src

函数描述与自动补全集成:

#+begin_src fish
function backup --description "backup a file"
    cp $argv[1] $argv[1].bak
end
#+end_src

** 返回值与状态码

#+begin_src fish
# fish 函数通过 return 返回退出码 (0=成功)
# 通过 echo 或 stdout 传递数据

function is_even
    if test (math $argv[1] % 2) -eq 0
        return 0    # 成功 (真)
    else
        return 1    # 失败 (假)
    end
end

if is_even 4
    echo "even"
end

# $status 是上一条命令的退出码
ls /nonexistent
echo $status    # 2
#+end_src
